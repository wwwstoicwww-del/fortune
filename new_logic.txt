    function calculateLifePath(y, m, d) {
        const sumString = (str) => String(str).split('').reduce((acc, curr) => acc + parseInt(curr), 0);
        let num = sumString(y) + sumString(m) + sumString(d);
        while (num > 9 && num !== 11 && num !== 22 && num !== 33) {
            num = sumString(num);
        }
        return num;
    }
